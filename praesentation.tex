\section{Geschichte von Wikiversity -- Standortbestimmung}
\subsection{Hintergrund: "`Open"' + E-Learning}
\begin{frame}
\frametitle{Hintergrund: "`Open"' + E-Learning}
\begin{itemize}
\item 1960er/1970er Jahre: \textbf{Open Source}, Hacker-Kultur
\item 1990er Jahre: \textbf{Open Access}
\item 2000er Jahre: \textbf{Open Science}, \textbf{Free Culture}, \textbf{Open~Education}
\end{itemize}

\begin{itemize}
\item \textbf{Digitale Medien}
\item E-Learning/Blended Learning
\end{itemize}
\end{frame}

\subsection{Entwicklung}
\begin{frame}
\frametitle{Entwicklung}
\begin{itemize}
\item \textbf{2001:} Wikipedia -- kollaborativ erstellte freie Inhalte
\item \textbf{2002:} MIT OpenCourseWare + UNESCO: "`OER"'
\item \textbf{2003:} Wikibooks mit Teilprojekt Wikiversity
\item \textbf{2004:} ZUM-Wiki
\item \textbf{2006:} Wikiversity, WikiEducator
\item \textbf{2006:} Aktivität habe "`stark nachgelassen"'
\item \textbf{2008\,ff.:} Wikiversity ist so gut wie tot
\item \textbf{Seitdem E-Learning-Boom:} iTunesU, MOOCs,\\\emph{flipped
    classroom}; Plattformen von Universitäten, Schulen, VHSen; Erfolg
  von Wikipedia
\item \textbf{2013:} AG Wikiversity
\end{itemize}
\end{frame}

\subsection{Zweckbestimmung von Wikiversity}
\begin{frame}
\frametitle{Zweckbestimmung von Wikiversity}
\textbf{Auszug aus dem \emph{Project proposal}:}
\vspace{.5em}

\begin{quote}
  \emph{Wikiversity is a centre for the creation and use of free
    learning materials and activities. Its primary priorities and
    goals are to:}
  \begin{itemize}
  \item \emph{\textbf{Create and host} a range of free-content,
      multilingual \textbf{learning materials}/resources, for all age
      groups in all languages}
  \item \emph{Host scholarly/learning projects and communities that
      \textbf{support these materials}}
  \item \emph{\textbf{Complement and develop existing Wikimedia
        projects} (eg. a project\\devoted to finding good sources for\\
      Wikipedia articles)}
  \end{itemize}
\end{quote}
\end{frame}

\subsection{Worum kümmert sich die AG Wikiversity?}
\begin{frame}
\frametitle{Worum kümmert sich die AG Wikiversity?}
\textbf{Vorschlag zu einem mehrgleisigen Ansatz:}
\vspace{1em}
\begin{itemize}
\item Strategie
\item Praxis
\item Community
\item Organisation
\item Technik
\item Finanzierung
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Strategie}
\textbf{Defizite und Probleme:}
\begin{itemize}
\item Diffuse Zielsetzung der Wikiversity, schlechtes Image
\item Kein Plan für die Entwicklung des Projekts
\item Unattraktiv für die Schule
\end{itemize}
\vspace{.5em}
\textbf{Mögliche Lösungen:}
\begin{itemize}
\item Konkreteres Profil als OER-Plattform
\item Langfristige Planung; Horizont: Fünf Jahre;\\Ziele, Mittel,
  Fristen, Partner festlegen
\item Relaunch der Wikiversity bekannt machen\\und dafür werben
\item Bereich Schule dem ZUM-Wiki überlassen
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Praxis 1}
\textbf{Defizite und Probleme:}
\begin{itemize}
\item Unattraktive Plattform: Inhalte nur im Benutzernamensraum,
  kryptische Seitennamen, nicht auffindbar, nicht wikifiziert, nicht
  kategorisiert
\end{itemize}
\vspace{.5em}
\textbf{Mögliche Lösungen:}
\begin{itemize}
\item Regeln zur Vereinheitlichung aufstellen und umsetzen
\item Wikiversity braucht eine QS
\begin{itemize}
\item Wikiversity insgesamt neu gestalten
\item Die Plattform kontinuierlich pflegen
\item Verschieben BNR $\rightarrow$ ANR
\item Inhalte auffindbar und wiederverwendbar\\machen
\item Abgebrochene Projekte und\\schlechte Inhalte löschen
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Praxis 2}
\textbf{Defizite und Probleme:}
\begin{itemize}
\item Wikiversity als Publikationsplattform für wissenschaftliche
  Arbeiten -- Seminar-, Bachelor-, Masterarbeiten; Aufsätze
\item Qualität?
\item Auffindbarkeit?
\end{itemize}
\vspace{.5em}
\textbf{Mögliche Lösungen:}
\begin{itemize}
\item Publikation nur unter Verantwortung der Dozenten
\item Mindestnote?
\item Sacherschließung: Kategorisieren
\item Metadaten: RDF, LRM, ZUM2Edutags
\item Aufnahme in die Suchmaschine BASE
\item Zeitschriften: Journal/Laboratory; Wikilogie
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Praxis 3}
\textbf{Defizite und Probleme:}
\begin{itemize}
\item Zahlreiche Verstöße gegen die Licensing Policy der WMF,
  Urheberrechtsverletzungen
\item Fehlendes Verständnis für OER und Urheberrecht bei Benutzern
\item Jahrelange Untätigkeit der Admins bei URVen
\item Überlastung der heutigen Admins mit der Pflege des Bestands
\end{itemize}
\vspace{.5em}
\textbf{Mögliche Lösungen:}
\begin{itemize}
\item Löschen von Dateien bei URVen
\item Deaktivieren des Uploads von Dateien
\item Strikte Befolgung der Licensing Policy der WMF
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Community}
\textbf{Defizite und Probleme:}
\begin{itemize}
\item Keine Community, die an gemeinsamen Zielen arbeitet (Dozenten,
  Studenten, Schüler, sonstige Interessierte)
\item Viel zu wenige Autoren und Admins
\item Fehlendes Interesse an Selbstverwaltung
\item Fehlendes Verständnis für das Arbeiten im Wiki und OER
\end{itemize}
\vspace{.5em}
\textbf{Mögliche Lösungen:}
\begin{itemize}
\item Werbung machen, einladen, einbinden, vernetzen
\item Schulungen, Workshops für Dozenten
\item Neue Richtlinien aufstellen und durchsetzen
\item Einbeziehen von OER/Wikiversity in\\GLAM-Projekte (Wikipedian in Residence)
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Organisation 1}
\textbf{Defizite und Probleme:}
\begin{itemize}
\item Die meisten "`Interessierten"' sind passiv
\item Die meisten Probleme von Wikiversity gehen auf die jahrelange
  Inaktivität der Admins zurück
\end{itemize}
\vspace{.5em}
\textbf{Mögliche Lösungen:}
\begin{itemize}
\item Klare Rollenverteilung zwischen den "`Interessierten"'
  vereinbaren
\item Arbeitsaufträge vergeben
\item Planung $\rightarrow$ Fristen setzen $\rightarrow$ kontrollieren
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Organisation 2}
\textbf{Defizite und Probleme:}
\begin{itemize}
\item Massiver Rückzug des deutschen Chapters aus dem Bildungsbereich
\item Sehr kurzfristige Planung (Jahrespläne)
\item Abschaffung des eigenen Referentennetzwerks
\end{itemize}
\vspace{.5em}
\textbf{Mögliche Lösungen:}
\begin{itemize}
\item Gründung einer eigenen Themorg für Bildung als Rahmen für alle
  Wikimedia-Bildungsaktivitäten
\item Selbstorganisiertes Referentennetzwerk mit\\selbstgesetzten
  Zielen
\item Zusammenarbeit mit dem Mentorenprogramm (Einführung des
  Gruppenmentoring)
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Technik}
\textbf{Defizite und Probleme:}
\begin{itemize}
\item Keine Einbindung von Multimedia-Inhalten von anderen Plattformen
  (v.\,a. YouTube)
\item Kein Upload von OpenDocument-Formaten möglich
\item Keine Kursverwaltung
\item Keine Metadaten für die Inhalte
\end{itemize}
\vspace{.5em}
\textbf{Mögliche Lösungen:}
\begin{itemize}
\item Installation von Multimedia-Plugins
\item Wikiversity-Moodle; ggf. auch Wikiversity-ILIAS
\item Mediawiki-Extensions: Education Program\\und Metadaten
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Finanzierung}
\textbf{Defizite und Probleme:}
\begin{itemize}
\item Tätigkeit als Admin auf Wikiversity ist Support;\\sehr aufwendig,
  weil sehr kleine Community
\item Schulungen für Dozenten/Studenten erforderlich
\end{itemize}
\vspace{.5em}
\textbf{Mögliche Lösungen:}
\begin{itemize}
\item Honorarkräfte übernehmen Support
\item Hochschulen zahlen für Schulungen und Support, der auf
  Wikiversity "`outgesourct"' wurde
\item Förderung durch WMF/WMDE\\(Community-Budget, Grants, Flow
  funding)
\item Einwerben von Drittmitteln
\end{itemize}
\end{frame}
